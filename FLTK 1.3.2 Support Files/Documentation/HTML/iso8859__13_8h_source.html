00077 static int
00078 iso8859_13_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
00079 {
00080   unsigned char c = 0;
00081   if (wc < 0x00a0) {
00082     *r = wc;
00083     return 1;
00084   }
00085   else if (wc >= 0x00a0 && wc < 0x0180)
00086     c = iso8859_13_page00[wc-0x00a0];
00087   else if (wc >= 0x2018 && wc < 0x2020)
00088     c = iso8859_13_page20[wc-0x2018];
00089   if (c != 0) {
00090     *r = c;
00091     return 1;
00092   }
00093   return RET_ILSEQ;
00094 }
