00221   class FL_EXPORT IntVector {
00222     int *arr;
00223     unsigned int _size;
00224     void init() {
00225       arr = NULL;
00226       _size = 0;
00227     }
00228     void copy(int *newarr, unsigned int newsize) {
00229       size(newsize);
00230       memcpy(arr, newarr, newsize * sizeof(int));
00231     }
00232   public:
00233     IntVector() { init(); }                                     // CTOR
00234     ~IntVector() { if ( arr ) free(arr); arr = NULL; }          // DTOR
00235     IntVector(IntVector&o) { init(); copy(o.arr, o._size); }    // COPY CTOR
00236     IntVector& operator=(IntVector&o) {                         // ASSIGN
00237       init();
00238       copy(o.arr, o._size);
00239       return(*this);
00240     }
00241     int operator[](int x) const { return(arr[x]); }
00242     int& operator[](int x) { return(arr[x]); }
00243     unsigned int size() { return(_size); }
00244     void size(unsigned int count) {
00245       if ( count != _size ) {
00246         arr = (int*)realloc(arr, count * sizeof(int));
00247         _size = count;
00248       }
00249     }
00250     int pop_back() { int tmp = arr[_size-1]; _size--; return(tmp); }
00251     void push_back(int val) { unsigned int x = _size; size(_size+1); arr[x] = val; }
00252     int back() { return(arr[_size-1]); }
00253   };

00453   void redraw_range(int topRow, int botRow, int leftCol, int rightCol) {
00454     if ( _redraw_toprow == -1 ) {
00455       // Initialize redraw range
00456       _redraw_toprow = topRow;
00457       _redraw_botrow = botRow;
00458       _redraw_leftcol = leftCol;
00459       _redraw_rightcol = rightCol;
00460     } else {
00461       // Extend redraw range
00462       if ( topRow < _redraw_toprow ) _redraw_toprow = topRow;
00463       if ( botRow > _redraw_botrow ) _redraw_botrow = botRow;
00464       if ( leftCol < _redraw_leftcol ) _redraw_leftcol = leftCol;
00465       if ( rightCol > _redraw_rightcol ) _redraw_rightcol = rightCol;
00466     }
00467     
00468     // Indicate partial redraw needed of some cells
00469     damage(FL_DAMAGE_CHILD);
00470   }
