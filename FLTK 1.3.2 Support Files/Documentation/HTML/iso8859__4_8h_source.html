00080 static int
00081 iso8859_4_wctomb (conv_t conv, unsigned char *r, ucs4_t wc, int n)
00082 {
00083   unsigned char c = 0;
00084   if (wc < 0x00a0) {
00085     *r = wc;
00086     return 1;
00087   }
00088   else if (wc >= 0x00a0 && wc < 0x0180)
00089     c = iso8859_4_page00[wc-0x00a0];
00090   else if (wc >= 0x02c0 && wc < 0x02e0)
00091     c = iso8859_4_page02[wc-0x02c0];
00092   if (c != 0) {
00093     *r = c;
00094     return 1;
00095   }
00096   return RET_ILSEQ;
00097 }
