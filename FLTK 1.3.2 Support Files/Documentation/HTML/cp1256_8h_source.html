00119     c = cp1256_page01[wc-0x0150];

00121     c = 0x88;

00125     c = cp1256_page20[wc-0x2008];
00126   else if (wc == 0x20ac)
00127     c = 0x80;
00128   else if (wc == 0x2122)
00129     c = 0x99;
