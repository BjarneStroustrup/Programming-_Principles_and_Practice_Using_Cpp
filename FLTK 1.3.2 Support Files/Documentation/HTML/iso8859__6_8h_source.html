00068 {
00069   unsigned char c = 0;
00070   if (wc < 0x00a0) {
00071     *r = wc;
00072     return 1;
00073   }
00074   else if (wc >= 0x00a0 && wc < 0x00b0)
00075     c = iso8859_6_page00[wc-0x00a0];
00076   else if (wc >= 0x0608 && wc < 0x0658)
00077     c = iso8859_6_page06[wc-0x0608];

00079     *r = c;
